```markdown
# TPS-OMS Architecture Decisions

## D-001: Monorepo — Next.js API Routes (not separate Express)
All API logic lives in Next.js route handlers. Avoids separate server management.
If complexity grows (background workers need long-running process), extract to separate service.

## D-002: Financial figures stored in paise (integer)
All monetary amounts stored as integers (paise = 1/100 rupee) to avoid float precision bugs.
Display: divide by 100, format with Intl.NumberFormat('en-IN').

## D-003: AES-256-GCM for FSSAI credentials
Each encrypted value stores IV + auth tag + ciphertext as base64-encoded JSON.
Key from ENCRYPTION_KEY env var. Credentials never appear in AuditLog value fields — only access events logged.

## D-004: Soft deletes only
No hard deletes on operational data. isActive/isDeleted flag used throughout.
AuthorityQuery and ProductList have NO delete route at all (permanent archive).

## D-005: Clock time stored as accumulated minutes per snapshot
StageLog stores employeeMinutes, clientMinutes, authorityMinutes as running totals.
Updated on every clock switch event. Timeline calculated from these totals, not from raw timestamps alone.
This survives server restarts and makes reporting fast.

## D-006: Project codes auto-generated
Format: TPS-YYYY-NNN (e.g., TPS-2024-001). Sequential per year. No gaps enforced.

## D-007: File storage — local for dev, S3 interface for prod
Documents saved to /uploads/ in dev. In prod, swap to S3 via env flag.
No file contents stored in DB — only URLs/paths.
```

```typescript
import { createCipheriv, createDecipheriv, randomBytes } from 'crypto'

const ALGORITHM = 'aes-256-gcm'
const KEY = Buffer.from(process.env.ENCRYPTION_KEY ?? '', 'hex')

export function encrypt(plaintext: string | null | undefined): string | null {
  if (!plaintext) return null
  const iv = randomBytes(12)
  const cipher = createCipheriv(ALGORITHM, KEY, iv)
  const encrypted = Buffer.concat([cipher.update(plaintext, 'utf8'), cipher.final()])
  const authTag = cipher.getAuthTag()
  return JSON.stringify({
    iv: iv.toString('hex'),
    tag: authTag.toString('hex'),
    data: encrypted.toString('hex'),
  })
}

export function decrypt(ciphertext: string | null | undefined): string | null {
  if (!ciphertext) return null
  const { iv, tag, data } = JSON.parse(ciphertext)
  const decipher = createDecipheriv(ALGORITHM, KEY, Buffer.from(iv, 'hex'))
  decipher.setAuthTag(Buffer.from(tag, 'hex'))
  return decipher.update(Buffer.from(data, 'hex')) + decipher.final('utf8')
}
```

```typescript
// INR formatting, date formatting, time elapsed display

export function formatRupees(paise: number): string {
  return new Intl.NumberFormat('en-IN', {
    style: 'currency',
    currency: 'INR',
    minimumFractionDigits: 2,
  }).format(paise / 100)
}

export function formatDate(date: Date | string | null | undefined): string {
  if (!date) return '—'
  return new Intl.DateTimeFormat('en-IN', {
    day: '2-digit',
    month: '2-digit',
    year: 'numeric',
  }).format(new Date(date))
}

export function formatTimeElapsed(minutes: number): string {
  if (minutes < 60) return `${minutes}m`
  const days = Math.floor(minutes / (60 * 24))
  const hours = Math.floor((minutes % (60 * 24)) / 60)
  const mins = minutes % 60
  const parts: string[] = []
  if (days > 0) parts.push(`${days}d`)
  if (hours > 0) parts.push(`${hours}h`)
  if (mins > 0) parts.push(`${mins}m`)
  return parts.join(' ') || '0m'
}

export function minutesSince(date: Date): number {
  return Math.floor((Date.now() - new Date(date).getTime()) / 60000)
}

export function getLicenseExpiryStatus(expiryDate: Date | null): 'green' | 'amber' | 'red' | 'none' {
  if (!expiryDate) return 'none'
  const daysUntil = Math.floor((new Date(expiryDate).getTime() - Date.now()) / 86400000)
  if (daysUntil > 90) return 'green'
  if (daysUntil > 30) return 'amber'
  return 'red'
}
```

```typescript
export async function generateProjectCode(): Promise<string> {
  const year = new Date().getFullYear()
  const prefix = `TPS-${year}-`
  const lastProject = await prisma.project.findFirst({
    where: { projectCode: { startsWith: prefix } },
    orderBy: { projectCode: 'desc' },

  const lastNum = lastProject
    ? parseInt(lastProject.projectCode.replace(prefix, ''), 10)
```

```typescript
import { ClockType } from '@prisma/client'
import { minutesSince } from './formatters'

interface ClockState {
  employeeMinutes: number
  clientMinutes: number
  authorityMinutes: number
  currentClock: ClockType
  clockSwitchedAt: Date
}

export function calculateAccumulatedMinutes(state: ClockState): {
  employeeMinutes: number
  clientMinutes: number
  authorityMinutes: number
} {
  const elapsed = minutesSince(state.clockSwitchedAt)
  return {
    employeeMinutes: state.employeeMinutes + (state.currentClock === 'EMPLOYEE' ? elapsed : 0),
    clientMinutes: state.clientMinutes + (state.currentClock === 'CLIENT' ? elapsed : 0),
    authorityMinutes: state.authorityMinutes + (state.currentClock === 'AUTHORITY' ? elapsed : 0),
  }
}

// Call this when switching clocks — persists the accumulated time to DB delta
export function accumulateClockSwitch(state: ClockState): {
  employeeMinutes: number
  clientMinutes: number
  authorityMinutes: number
} {
  return calculateAccumulatedMinutes(state)
}
```

```typescript
import { NextAuthOptions } from 'next-auth'
import CredentialsProvider from 'next-auth/providers/credentials'
import bcrypt from 'bcryptjs'
import { prisma } from './prisma'
import { Role } from '@prisma/client'

export const authOptions: NextAuthOptions = {
  providers: [
    CredentialsProvider({
      name: 'credentials',
      credentials: {
        email: { label: 'Email', type: 'email' },
        password: { label: 'Password', type: 'password' },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) return null
        const user = await prisma.user.findUnique({
          where: { email: credentials.email },
        })
        if (!user || !user.isActive) return null
        const valid = await bcrypt.compare(credentials.password, user.password)
        if (!valid) return null
        return { id: user.id, name: user.name, email: user.email, role: user.role }
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id
        token.role = (user as any).role as Role
      }
      return token
    },
    async session({ session, token }) {
      session.user.id = token.id as string
      session.user.role = token.role as Role
      return session
    },
  },
  pages: {
    signIn: '/login',
  },
  session: {
    strategy: 'jwt',
    maxAge: 30 * 60, // 30 minutes
  },
}
```

- [ ] **Step 2: Create `app/api/licenses/[id]/credentials/route.ts`** (most security-critical endpoint)

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { getServerSession } from 'next-auth'
import { authOptions } from '@/lib/auth'
import { prisma } from '@/lib/prisma'
import { decrypt } from '@/lib/encryption'
import { writeAuditLog } from '@/lib/audit'
import { ROLES_WITH_CREDENTIAL_ACCESS } from '@/types'

export async function GET(
  req: NextRequest,
  { params }: { params: { id: string } }
) {
  const session = await getServerSession(authOptions)
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  // Role check — must be MANAGER, DIRECTOR, or SUPER_ADMIN
  // Executives can only see credentials for their own assigned projects' licenses
  if (!ROLES_WITH_CREDENTIAL_ACCESS.includes(session.user.role as any)) {
    // Check if executive has an active project on this license
    const license = await prisma.license.findUnique({ where: { id: params.id } })
    if (!license) return NextResponse.json({ error: 'Not found' }, { status: 404 })
    const project = await prisma.project.findFirst({
      where: { licenseId: params.id, assignedToId: session.user.id, status: 'ACTIVE' },
    })
    if (!project) return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
  }

  const license = await prisma.license.findUnique({
    where: { id: params.id },
    select: { portalUsername: true, portalPassword: true },
  })
  if (!license) return NextResponse.json({ error: 'Not found' }, { status: 404 })

  // Log the reveal action — never log the actual credential values
  await writeAuditLog({
    userId: session.user.id,
    action: 'CREDENTIAL_REVEALED',
    entityType: 'License',
    entityId: params.id,
    ipAddress: req.headers.get('x-forwarded-for') || undefined,
  })

  return NextResponse.json({
    username: license.portalUsername,
    password: decrypt(license.portalPassword),
  })
}
```

```typescript
// POST — mark stage complete, start next stage
// 1. Snapshot accumulated minutes onto current stage
// 2. Set completedAt, completedById, status=COMPLETED
// 3. Find next stage from workflow template
// 4. Create new StageLog for next stage with EMPLOYEE clock
// 5. Create notification to manager if stage is isApprovalGate
export async function POST(req, { params }) {
  const session = await getServerSession(authOptions)
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const stage = await prisma.stageLog.findUnique({ where: { id: params.stageId } })
  if (!stage) return NextResponse.json({ error: 'Not found' }, { status: 404 })

  const { body } = await req.json().catch(() => ({ body: {} }))
  const accumulated = accumulateClockSwitch(stage as any)

  await prisma.stageLog.update({
    where: { id: params.stageId },
    data: {
      status: 'COMPLETED',
      completedAt: new Date(),
      completedById: session.user.id,
      ...accumulated,
      notes: body.notes,
    },
  })

  // Get project + service type to find next stage
  const project = await prisma.project.findUnique({
    where: { id: params.id },
    include: { serviceType: { include: { workflowTemplate: { orderBy: { stageNumber: 'asc' } } } } },
  })
  const nextStageTemplate = project?.serviceType.workflowTemplate.find(
    t => t.stageNumber === stage.stageNumber + 1
  )

  if (nextStageTemplate) {
    await prisma.stageLog.create({
      data: {
        projectId: params.id,
        stageNumber: nextStageTemplate.stageNumber,
        stageName: nextStageTemplate.stageName,
        status: 'IN_PROGRESS',
        currentClock: 'EMPLOYEE',
        clockSwitchedAt: new Date(),
      },
    })
  }

  return NextResponse.json({ ok: true })
}
```

```typescript
      status: 'PENDING',

  })

  // Notify all managers

      `${session.user.name} has requested a block on project ${params.id}`,
```

```typescript
      : calculateAccumulatedMinutes(stage as any)
    const total = live.employeeMinutes + live.clientMinutes + live.authorityMinutes
    return { ...stage, ...live, totalMinutes: total }
  })

  const totals = result.reduce((acc, s) => ({
    totalMinutes: acc.totalMinutes + s.totalMinutes,
    totalEmployeeMinutes: acc.totalEmployeeMinutes + s.employeeMinutes,
    totalClientMinutes: acc.totalClientMinutes + s.clientMinutes,
    totalAuthorityMinutes: acc.totalAuthorityMinutes + s.authorityMinutes,
  }), { totalMinutes: 0, totalEmployeeMinutes: 0, totalClientMinutes: 0, totalAuthorityMinutes: 0 })

  return NextResponse.json({ stages: result, ...totals })
}
```

```tsx
'use client'
import { useState, useEffect } from 'react'
import { formatTimeElapsed, minutesSince } from '@/lib/formatters'
import { ClockType } from '@prisma/client'

interface TimeElapsedProps {
  accumulatedMinutes: number
  clockSwitchedAt: Date | string
  currentClock: ClockType
  isCompleted?: boolean
}

export function TimeElapsed({ accumulatedMinutes, clockSwitchedAt, currentClock, isCompleted }: TimeElapsedProps) {
  const [total, setTotal] = useState(accumulatedMinutes + (isCompleted ? 0 : minutesSince(clockSwitchedAt)))

  useEffect(() => {
    if (isCompleted) return
    const interval = setInterval(() => {
      setTotal(accumulatedMinutes + minutesSince(clockSwitchedAt))
    }, 60000)
    return () => clearInterval(interval)
  }, [accumulatedMinutes, clockSwitchedAt, isCompleted])

  return <span>{formatTimeElapsed(total)}</span>
}
```

```tsx
  projectId: string
  stageId: string

  onSuccess: () => void

export function BlockingRequestModal({ open, onClose, projectId, stageId, blockType, onSuccess }: Props) {
  const [reason, setReason] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')

  async function submit() {

      method: 'POST',
      headers: { 'Content-Type': 'application/json' },

    })
    setLoading(false)
    if (res.ok) { onSuccess(); onClose() }
    else setError('Failed to submit request')
  }

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Request Block — Pending with {blockType === 'CLIENT' ? 'Client' : 'Authority'}</DialogTitle>
        </DialogHeader>

          </p>

            <Label>Reason (required)</Label>
            <Textarea
              value={reason}
              onChange={e => setReason(e.target.value)}
              placeholder="e.g. Signed draft pending from client — sent on 01/07/2024"
              rows={3}
            />

          {error && <p className="text-sm text-red-600">{error}</p>}
```

```typescript
    { days: 30, type: 'LICENSE_EXPIRY_30' as const },
  ]

  for (const { days, type } of thresholds) {
    const targetDate = new Date(now.getTime() + days * 86400000)
    const licenses = await prisma.license.findMany({
      where: {
        status: 'ACTIVE',
        renewalAlerts: true,
        expiryDate: {
          gte: new Date(targetDate.getTime() - 86400000),
          lte: targetDate,
        },
      },
      include: { client: true },
    })

    for (const license of licenses) {
      const managers = await prisma.user.findMany({
        where: { role: { in: days <= 30 ? ['MANAGER', 'DIRECTOR', 'SUPER_ADMIN'] : ['MANAGER', 'DIRECTOR', 'SUPER_ADMIN'] }, isActive: true },
      })
      for (const m of managers) {
        await createNotification(m.id, type,
          `License Expiring in ${days} Days`,
          `${license.client.legalName} — License ${license.licenseNumber || 'N/A'} expires in ${days} days.`
        )
      }
    }
  }
}

async function checkAnnualReturn() {
  const now = new Date()
  const isApril1 = now.getMonth() === 3 && now.getDate() === 1
  const isMay15 = now.getMonth() === 4 && now.getDate() === 15
  if (!isApril1 && !isMay15) return

  const activeLicenses = await prisma.license.findMany({ where: { status: 'ACTIVE' } })
  const managers = await prisma.user.findMany({
    where: { role: { in: ['MANAGER', 'DIRECTOR', 'SUPER_ADMIN'] }, isActive: true },
  })

  for (const m of managers) {
    await createNotification(m.id, 'ANNUAL_RETURN_REMINDER',
      isApril1 ? 'Annual Return Window Open' : '⚠️ Annual Return Deadline Approaching (May 31)',
      `${activeLicenses.length} active licenses require Annual Return filing.`
    )
  }
}

async function checkOverdueStages() {
  // Find IN_PROGRESS stages where employee clock has been running past SLA
  // Get SLA from workflow template for that stage
  const activeStages = await prisma.stageLog.findMany({
    where: { status: 'IN_PROGRESS', currentClock: 'EMPLOYEE' },
    include: { project: { include: { serviceType: { include: { workflowTemplate: true } }, assignedTo: true } } },
  })

  for (const stage of activeStages) {
    const template = stage.project.serviceType.workflowTemplate.find(t => t.stageNumber === stage.stageNumber)
    if (!template) continue
    const slaMinutes = template.slaDays * 24 * 60
    const elapsed = stage.employeeMinutes + minutesSinceDate(stage.clockSwitchedAt)
    if (elapsed > slaMinutes && stage.project.assignedToId) {
      await createNotification(
        stage.project.assignedToId,
        'STAGE_OVERDUE',
        'Stage Overdue',
        `Stage "${stage.stageName}" on project ${stage.project.projectCode} is past SLA.`,
        stage.projectId
      )
    }
  }
}

function minutesSinceDate(date: Date): number {
  return Math.floor((Date.now() - new Date(date).getTime()) / 60000)
}

async function checkOverduePayments() {
  const thirtyDaysAgo = new Date(Date.now() - 30 * 86400000)
  const overduePayments = await prisma.payment.findMany({
    where: {
      balancePending: { gt: 0 },
      project: { status: 'COMPLETED', completedDate: { lte: thirtyDaysAgo } },
    },
    include: { project: { include: { assignedTo: true } } },
  })

  for (const p of overduePayments) {
    if (p.project.assignedToId) {
      await createNotification(p.project.assignedToId, 'PAYMENT_OUTSTANDING',
        'Payment Outstanding — Closed Project',
        `Project ${p.project.projectCode} has been closed for 30+ days with payment pending.`,
        p.projectId
      )
    }
  }
}
```

```typescript
// GET /api/employees/[id]/performance?from=YYYY-MM-DD&to=YYYY-MM-DD
// Returns projects completed in range with:
// - fairEmployeeTime (sum of employee clock minutes across all stages)
// - clientTime, authorityTime

// - rejectedBlockCount (anti-abuse flag if >3 in period)

const stages = await prisma.stageLog.findMany({
  where: {
    completedById: params.id,
    completedAt: { gte: new Date(from), lte: new Date(to) },
    project: { status: 'COMPLETED' },
  },
  include: {
    project: { include: { serviceType: { include: { workflowTemplate: true } } } },
  },
})

const score = stages.reduce((acc, stage) => {
  const template = stage.project.serviceType.workflowTemplate.find(t => t.stageNumber === stage.stageNumber)
  const targetMinutes = (template?.slaDays || 1) * 24 * 60
  return {
    fairEmployeeMinutes: acc.fairEmployeeMinutes + stage.employeeMinutes,
    targetMinutes: acc.targetMinutes + targetMinutes,
  }
}, { fairEmployeeMinutes: 0, targetMinutes: 0 })

const performanceScore = score.fairEmployeeMinutes > 0
  ? Math.round((score.targetMinutes / score.fairEmployeeMinutes) * 100)
  : 100
```

5. No DELETE route for: `AuthorityQuery`, `ProductList`, `ProductItem`, `AuditLog`.
6. All money as `paise: Int`. Display: `paise / 100` with `Intl.NumberFormat('en-IN')`.
